CREATE DATABASE ONLINEBOOKSTORE;
USE ONLINEBOOKSTORE;
CREATE TABLE AUTHORS(AUTHOR_ID INT PRIMARY KEY,AUTHOR_NAME VARCHAR(20),COUNTRY VARCHAR(20));
CREATE TABLE BOOKS( BOOK_ID INT PRIMARY KEY,AUTHOR_ID INT ,TITLE VARCHAR(20),PRICE DECIMAL(10,2),PUBLICATION_YEAR DATE,FOREIGN KEY(AUTHOR_ID) REFERENCES AUTHORS(AUTHOR_ID));
CREATE TABLE ORDERS(ORDER_ID INT PRIMARY KEY,BOOK_ID INT ,CUSTOMER_NAME VARCHAR(20),ORDER_DATE DATE,FOREIGN KEY(BOOK_ID)REFERENCES BOOKS(BOOK_ID));
